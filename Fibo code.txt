
#include <iostream>

void main()
{
	int a[50];% sequence number is 50
	a[0]=0;
	a[1]=1;
	for(i=2;i<50;i++)
	{
	a[i]=a[i-1]+a[i-2];
	}
}



%recursio method
#include <iostream>
int fibo(int n)
{
	if(n==0)
	return 0;
	else if(n==1)
	return 1;
	else return fibo(n-1)+fibo(n-2);
}
int main()
{
	int n;
	cin>>n;% input sequence number
	a[n];
	for(int i=0;i<n;i++)
	a[i]=fibo(i);
}
